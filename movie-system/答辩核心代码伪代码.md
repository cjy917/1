### 2. useAnalytics.js（组合式函数）

```javascript
// 关键代码1: 导入API服务
import { analyticsApi } from '../services/analytics'

// 关键代码2: 响应式状态定义
const overview = ref(null)
const genres = ref([])
const loading = ref(false)

// 关键代码3: 批量并行请求（核心！）
async function fetchDashboard(filters) {
  loading.value = true
  // 16个接口并行请求，提高加载速度
  const results = await Promise.all([
    analyticsApi.getOverview(filters),
    analyticsApi.getGenres(15, filters),
    analyticsApi.getYears(filters),
    // ... 共16个接口
  ])
  overview.value = results[0]
  genres.value = results[1]
  // ... 按顺序赋值
}

// 关键代码4: 对外暴露
export function useAnalytics() {
  return { overview, genres, loadAllData: fetchDashboard }
}
```

**老师可能问：**
- Q: 为什么用 Promise.all 而不是逐个请求？
- A: 并行请求减少总耗时，16个接口同时请求只需等最慢的那个
- Q: 数据怎么缓存的？
- A: 用 hasLoaded 和 cacheVersion 判断，离开页面再回来不重复加载

### 3. analytics.js（API服务层）

```javascript
// 关键代码1: 创建axios实例，配置代理路径
const api = axios.create({
  baseURL: '/api/analytics',  // ← Vite代理会转发到后端
  timeout: 30000

export const analyticsApi = {
  getOverview(filters = {}) {
    // 实际请求: GET /api/analytics/overview?genre=Action
    return api.get('/overview', { params: filters }).then(res => res.data)
  },
  getGenres(limit = 15, filters = {}) {
    return api.get('/genres', { params: { limit, ...filters } }).then(res => res.data)
  },
  // ... 共19个方法
}
```

**老师可能问：**
- Q: 前端怎么调用后端的？

- Q: 为什么不用完整URL（http://127.0.0.1:5000）？
- A: 用相对路径，Vite开发服务器会自动代理到后端（vite.config.js 配置了 proxy）

### 5. analytics_service.py（业务逻辑层）

```python
# 关键代码1: 导入数据库连接
from services.movie_service import get_mysql

# 关键代码2: 解析筛选参数（核心！）
def _parse_filters(genre=None, year=None, country=None):
    conditions = []
    params = []
    if genre:
        # 处理多值类型筛选: "剧情,喜剧" → LIKE %剧情% OR LIKE %喜剧%
        genre_list = [g.strip() for g in genre.split(",")]
        parts = ["genres LIKE %s" for _ in genre_list]
        params.extend(f"%{g}%" for g in genre_list)
        conditions.append("(" + " OR ".join(parts) + ")")
    # ... 处理 year、country
    return " AND ".join(conditions), params

# 关键代码3: 概览统计核心逻辑
def overview_stats(genre=None, year=None, country=None, user_count=0, rating_count=0):
    where_clause, params = _parse_filters(genre, year, country)
    conn = get_mysql()
    cursor = conn.cursor()
    # 统计电影总数
    sql = "SELECT COUNT(*) FROM movies" + (" WHERE " + where_clause if where_clause else "")
    cursor.execute(sql, params)
    total_movies = cursor.fetchone()[0]
    cursor.close()
    conn.close()
    # 返回字典（前端直接使用）
    return {
        "total_movies": total_movies,
        "total_users": user_count,
        "total_ratings": rating_count,
    }

# 关键代码4: 类型分布核心逻辑
def genre_distribution(limit=15, **filters):
    # 步骤1: 解析筛选条件 → WHERE子句
    where_clause, params = _parse_filters(**filters)
    
    # 步骤2: 查询所有genres字段（用管道符|分隔多个类型）
    sql = "SELECT genres FROM movies WHERE {where} AND genres IS NOT NULL AND genres != ''"
    cursor.execute(sql, params)
    
    # 步骤3: Python端分割统计（MySQL不支持数组展开，用split_pipe处理）
    counter = Counter()
    for row in cursor.fetchall():
        # genres = "剧情|喜剧|爱情" → ["剧情", "喜剧", "爱情"]
        for name in split_pipe(row["genres"]):
            counter[name] += 1
    
    # 步骤4: 返回TopN结果
    return [{"name": name, "count": count} for name, count in counter.most_common(limit)]
```

**老师可能问：**
- Q: 筛选参数怎么变成 SQL 的？
- A: _parse_filters() 把 genre="Action" 转成 "genres LIKE %Action%"
- Q: 一部电影有多个类型怎么统计的？
- A: genres字段用\|分隔存储（如"剧情|喜剧|爱情"），查询后用split_pipe()在Python端分割，再用Counter统计
- Q: 为什么不在SQL里直接处理？
- A: MySQL不支持数组展开函数（如PostgreSQL的UNNEST），所以在Python端处理
- Q: 为什么用 get_mysql() 而不是直接连数据库？
- A: 封装了连接池，统一管理数据库连接，避免重复创建连接

## 三、筛选功能完整流程（老师常问）

```
用户在筛选栏选择: 类型=剧情, 年份=2020
         ↓
FilterBar → handleFiltersDraft() 更新本地状态
         ↓
用户点击搜索 → handleSearch()
         ↓
loadFilteredData({ genre: "剧情", year: "2020" })
         ↓
useAnalytics.js → fetchDashboard(filters)
         ↓
analytics.js → axios.get('/overview', { params: { genre: "剧情", year: "2020" } })
         ↓
实际请求: GET /api/analytics/overview?genre=剧情&year=2020
         ↓
app.py → analytics_filters() → {"genre": "剧情", "year": "2020"}
         ↓
overview_stats(**filters) → overview_stats(genre="剧情", year="2020")
         ↓
_parse_filters() → "genres LIKE %剧情% AND year = 2020"
         ↓
SQL查询 → 返回筛选后的统计数据 → jsonify() → 前端 → 更新图表
```
